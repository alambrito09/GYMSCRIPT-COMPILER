ESTRUCTURA DE UN PROGRAMA

rutina NombreRutina {
  reps variable = valor;
  peso decimal  = valor;
  serie texto   = "algo";
  descanso flag = verdad;

  si_cumplo (condición) {
    levantar(variable);
  } sino {
    levantar("otro");
  }

  mientras_pueda (condición) {
    // instrucciones
  }

  retornar resultado;
}

REGLAS DE ESCRITURA

• Sensible a mayúsculas/minúsculas
• Identificadores: letra/_ + letras/dígitos/_
• Toda instrucción termina en ;
• Bloques delimitados con { }
• Cadenas con comillas dobles "..."
• Comentarios con //
• Indentación recomendada: 2 espacios
• Nombres de rutina: empiezan con mayúscula

BNF / EBNF DEL LENGUAJE GYMSCRIPT

<programa> ::= { <rutina> }
<rutina> ::= "rutina" <IDENTIFICADOR> "{" <instrucciones> "}"
<instrucciones> ::= { <instruccion> }
<instruccion> ::= <declaracion> | <asignacion> | <imprimir> | <condicional> | <bucle_while> | <retorno>

<declaracion> ::= <tipo> <IDENTIFICADOR> "=" <expresion> ";"
<tipo> ::= "reps" | "peso" | "serie" | "descanso"
<asignacion> ::= <IDENTIFICADOR> "=" <expresion> ";"
<imprimir> ::= "levantar" "(" <expresion> ")" ";"
<condicional> ::= "si_cumplo" "(" <expresion> ")" "{" <instrucciones> "}" [ "sino" "{" <instrucciones> "}" ]
<bucle_while> ::= "mientras_pueda" "(" <expresion> ")" "{" <instrucciones> "}"
<retorno> ::= "retornar" <expresion> ";"

<expresion> ::= <termino> { ("+" | "-" | "==" | "!=" | "<" | ">" | "<=" | ">=" | "&&" | "||") <termino> }
<termino> ::= <factor> { ("*" | "/") <factor> }
<factor> ::= <NUMERO> | <DECIMAL> | <CADENA> | <BOOL> | <IDENTIFICADOR> | "(" <expresion> ")" | "!" <factor>

<IDENTIFICADOR> ::= [a-zA-Z_][a-zA-Z0-9_]*
<NUMERO> ::= [0-9]+
<DECIMAL> ::= [0-9]+\.[0-9]+
<CADENA> ::= "\"" [^"]* "\""
<BOOL> ::= "verdad" | "falso"
